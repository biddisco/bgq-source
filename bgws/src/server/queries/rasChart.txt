-- RAS_CHART

SELECT CURRENT_DATE - 4 DAYS AS d1,
      COALESCE( SUM(CASE WHEN severity = 'FATAL' AND DATE(event_time) = CURRENT_DATE - 4 DAYS THEN 1 ELSE 0 END), 0 ) AS f1,
      COALESCE( SUM(CASE WHEN severity = 'WARN' AND DATE(event_time) = CURRENT_DATE - 4 DAYS THEN 1 ELSE 0 END), 0 ) AS w1,
      COALESCE( SUM(CASE WHEN severity = 'INFO' AND DATE(event_time) = CURRENT_DATE - 4 DAYS THEN 1 ELSE 0 END), 0 ) AS i1,
      CURRENT_DATE - 3 DAYS AS d2,
      COALESCE( SUM(CASE WHEN severity = 'FATAL' AND DATE(event_time) = CURRENT_DATE - 3 DAYS THEN 1 ELSE 0 END), 0 ) AS f2,
      COALESCE( SUM(CASE WHEN severity = 'WARN' AND DATE(event_time) = CURRENT_DATE - 3 DAYS THEN 1 ELSE 0 END), 0 ) AS w2,
      COALESCE( SUM(CASE WHEN severity = 'INFO' AND DATE(event_time) = CURRENT_DATE - 3 DAYS THEN 1 ELSE 0 END), 0 ) AS i2,
      CURRENT_DATE - 2 DAYS AS d3,
      COALESCE( SUM(CASE WHEN severity = 'FATAL' AND DATE(event_time) = CURRENT_DATE - 2 DAYS THEN 1 ELSE 0 END), 0 ) AS f3,
      COALESCE( SUM(CASE WHEN severity = 'WARN' AND DATE(event_time) = CURRENT_DATE - 2 DAYS THEN 1 ELSE 0 END), 0 ) AS w3,
      COALESCE( SUM(CASE WHEN severity = 'INFO' AND DATE(event_time) = CURRENT_DATE - 2 DAYS THEN 1 ELSE 0 END), 0 ) AS i3,
      CURRENT_DATE - 1 DAYS AS d4,
      COALESCE( SUM(CASE WHEN severity = 'FATAL' AND DATE(event_time) = CURRENT_DATE - 1 DAYS THEN 1 ELSE 0 END), 0 ) AS f4,
      COALESCE( SUM(CASE WHEN severity = 'WARN' AND DATE(event_time) = CURRENT_DATE - 1 DAYS THEN 1 ELSE 0 END), 0 ) AS w4,
      COALESCE( SUM(CASE WHEN severity = 'INFO' AND DATE(event_time) = CURRENT_DATE - 1 DAYS THEN 1 ELSE 0 END), 0 ) AS i4,
      CURRENT_DATE AS d5,
      COALESCE( SUM(CASE WHEN severity = 'FATAL' AND DATE(event_time) = CURRENT_DATE THEN 1 ELSE 0 END), 0 ) AS f5,
      COALESCE( SUM(CASE WHEN severity = 'WARN' AND DATE(event_time) = CURRENT_DATE THEN 1 ELSE 0 END), 0 ) AS w5,
      COALESCE( SUM(CASE WHEN severity = 'INFO' AND DATE(event_time) = CURRENT_DATE THEN 1 ELSE 0 END), 0 ) AS i5
 FROM bgqEventLog
 WHERE event_time > CURRENT_DATE - 5 DAYS
